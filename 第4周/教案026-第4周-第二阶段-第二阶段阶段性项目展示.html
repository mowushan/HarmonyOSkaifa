📝 资产录入页参考

@Entry
@Component
struct AssetInputPage {
  @State assetName: string = ''
  @State assetPrice: string = ''
  
  build() {
    Column({ space: 20 }) {
      Text('资产录入')
        .fontSize(24)
        .fontWeight(FontWeight.Bold)
        .margin({ top: 20 })
      
      TextInput({ placeholder: '请输入资产名称' })
        .onChange((value) => { this.assetName = value })
      
      TextInput({ placeholder: '请输入资产价格' })
        .type(InputType.Number)
        .onChange((value) => { this.assetPrice = value })
      
      Button('提交')
        .width('100%')
        .backgroundColor('#0078d4')
        .onClick(() => {
          console.log('资产名称:', this.assetName)
          console.log('资产价格:', this.assetPrice)
        })
    }
    .padding(20)
    .width('100%')
    .height('100%')
  }
}

📋 资产列表页参考

@Entry
@Component
struct AssetListPage {
  @State assetList: string[] = ['笔记本电脑', '投影仪', '交换机']
  
  build() {
    Column() {
      Text('资产列表')
        .fontSize(24)
        .fontWeight(FontWeight.Bold)
        .margin(20)
      
      List() {
        ForEach(this.assetList, (item: string, index: number) => {
          ListItem() {
            Row() {
              Text(item)
                .fontSize(18)
                .margin({ left: 10 })
              Blank()
              Text('详情')
                .fontColor('#0078d4')
                .onClick(() => {
                  console.log('点击:', item)
                })
            }
            .padding(12)
            .width('100%')
          }
          .backgroundColor('#f5f5f5')
          .borderRadius(12)
          .margin({ top: 8 })
        })
      }
      .padding(20)
      .width('100%')
    }
  }
}

    🔧 现场优化工作坊 · 让作品更完美

    💡 常见问题与优化建议

    🎨 界面美观优化

    问题： 组件间距不均匀，颜色刺眼
    优化： 使用统一的margin/padding，参考鸿蒙设计规范颜色。

    // 统一卡片样式
.card() {
  .backgroundColor('#ffffff')
  .borderRadius(16)
  .padding(16)
  .margin({ top: 12, left: 16, right: 16 })
  .shadow({ radius: 8, color: '#cccccc' })
}

    📱 布局响应式优化

    问题： 在大屏幕或折叠屏上布局错乱
    优化： 使用百分比宽度或Flex布局的权重。

    // 响应式宽度
Row() {
  Text('名称').width('30%')
  TextInput().width('70%')
}

    ⚡ 性能小优化

    问题： 列表滑动卡顿
    优化： List组件使用LazyForEach替代ForEach加载大数据。

    // 懒加载列表
LazyForEach(this.dataSource, 
  (item: string) => {
    ListItem() { ... }
  }
)
