2. 数据查询：从数据库拿数据

// 数据库帮助类中的查询方法示例 public async queryAllAssets(): Promise<Array<AssetItem>> { let resultSet: relationalStore.ResultSet = await this.store.querySql( "SELECT * FROM asset_table ORDER BY id DESC" ); let assets: AssetItem[] = []; while (resultSet.goToNextRow()) { let asset: AssetItem = { id: resultSet.getLong(resultSet.getColumnIndex("id")), name: resultSet.getString(resultSet.getColumnIndex("name")), category: resultSet.getString(resultSet.getColumnIndex("category")), quantity: resultSet.getLong(resultSet.getColumnIndex("quantity")), location: resultSet.getString(resultSet.getColumnIndex("location")), purchaseDate: resultSet.getString(resultSet.getColumnIndex("purchase_date")) }; assets.push(asset); } resultSet.close(); return assets; }

3. 列表展示：List + ForEach

// 资产列表页核心代码结构 @Entry @Component struct AssetListPage { @State assets: AssetItem[] = []; aboutToAppear() { this.loadAssets(); } async loadAssets() { let dbHelper = new DatabaseHelper(); this.assets = await dbHelper.queryAllAssets(); } build() { Column() { if (this.assets.length === 0) { // 空状态占位 Text('暂无资产数据，请先添加') .fontSize(16) .fontColor('#999') .margin({ top: 50 }) } else { List() { ForEach(this.assets, (item: AssetItem) => { ListItem() { // 每一项的UI AssetListItem({ asset: item }) } }, (item: AssetItem) => item.id.toString()) } .width('100%') .height('100%') } } } }
