📝 考核任务 · 资产管理系统界面开发

🏗️ 项目需求

开发一个简易的“固定资产管理系统”前端界面，包含以下三个页面，并实现页面间跳转：

  1. 资产录入页 (AddAssetPage)：表单输入资产名称、资产编号、购入日期、价格，点击“提交”按钮后跳转到资产列表页，并将数据传递过去。
  2. 资产列表页 (AssetListPage)：展示所有已添加的资产（卡片形式），每条数据包含名称、编号、价格。点击某一条资产，跳转到详情页并传递该资产数据。
  3. 资产详情页 (AssetDetailPage)：展示选中资产的完整信息（名称、编号、日期、价格），并提供一个“返回列表”按钮。

📌 额外加分项（最多+10分）： 实现表单验证（例如资产名称不能为空、价格必须为正数）或增加删除资产的功能。

export interface Asset {
    id: string;          // 唯一标识（可用时间戳）

    🔰 阶段三：开发资产录入页（20分钟）

    📄 AddAssetPage.ets 核心代码参考（含表单验证）：

    import router from '@ohos.router';
import { Asset, generateId } from '../model/AssetModel';

@Entry
@Component
struct AddAssetPage {
    @State assetName: string = '';
    @State assetCode: string = '';
    @State buyDate: string = '';
    @State priceStr: string = '';
    @State nameError: string = '';
    @State priceError: string = '';

    // 表单验证
    validateForm(): boolean {
        let valid = true;
        if (!this.assetName.trim()) {
            this.nameError = '资产名称不能为空';
            valid = false;
        } else {
            this.nameError = '';
        }
        const priceNum = parseFloat(this.priceStr);
        if (isNaN(priceNum) || priceNum <= 0) {
            this.priceError = '价格必须是正数';
            valid = false;
        } else {
            this.priceError = '';
        }
        return valid;
    }

    // 提交资产
    submitAsset() {
        if (!this.validateForm()) return;

        const newAsset: Asset = {
            id: generateId(),
            name: this.assetName,
            code: this.assetCode,
            date: this.buyDate,
            price: parseFloat(this.priceStr)
        };
        // 跳转到列表页，并通过params传递新增的资产
        router.pushUrl({
            url: 'pages/AssetListPage',
            params: { newAsset: newAsset }
        });
    }

    build() {
        Column({ space: 20 }) {
            Text('➕ 添加固定资产')
                .fontSize(24).fontWeight(FontWeight.Bold).margin({ top: 20 })
            
            // 资产名称输入框
            TextInput({ placeholder: '资产名称', text: this.assetName })
                .onChange((value: string) => { this.assetName = value })
            if (this.nameError) {
                Text(this.nameError).fontSize(12).fontColor('#FF0000')
            }
            
            // 资产编号输入框
            TextInput({ placeholder: '资产编号', text: this.assetCode })
                .onChange((value: string) => { this.assetCode = value })
            
            // 购入日期
            TextInput({ placeholder: '购入日期 (2025-03-20)', text: this.buyDate })
                .onChange((value: string) => { this.buyDate = value })
            
            // 价格输入框
            TextInput({ placeholder: '价格 (元)', text: this.priceStr })
                .type(InputType.Number)
                .onChange((value: string) => { this.priceStr = value })
            if (this.priceError) {
                Text(this.priceError).fontSize(12).fontColor('#FF0000')
            }
            
            Button('提交资产', { type: ButtonType.Capsule })
                .width('80%').backgroundColor('#0078D4')
                .onClick(() => this.submitAsset())
            
            Blank().layoutWeight(1)
        }
        .padding(20)
        .width('100%')
        .height('100%')
    }
}

    🔰 阶段四：开发资产列表页（20分钟）

    📄 AssetListPage.ets 核心代码参考（接收新增 + 跳转详情）：

    import router from '@ohos.router';
import { Asset } from '../model/AssetModel';

@Entry
@Component
struct AssetListPage {
    @State assetList: Asset[] = [];

    aboutToAppear() {
        // 接收从录入页传来的新资产
        const params = router.getParams() as Record;
        if (params && params['newAsset']) {
            const newAsset = params['newAsset'] as Asset;
            this.assetList = [newAsset, ...this.assetList];
        }
        // 实际项目中这里会从本地数据库加载，本次考核演示内存存储
    }

    // 跳转到详情页
    goToDetail(asset: Asset) {
        router.pushUrl({
            url: 'pages/AssetDetailPage',
            params: { assetDetail: asset }
        });
    }

    build() {
        Column() {
            Text('📋 资产清单')
                .fontSize(24).fontWeight(FontWeight.Bold).margin(15)
            
            if (this.assetList.length === 0) {
                Text('暂无资产，请先添加 ➕')
                    .fontSize(16).fontColor('#999').margin({ top: 50 })
            } else {
                List() {
                    ForEach(this.assetList, (item: Asset) => {
                        ListItem() {
                            Column() {
                                Text(item.name).fontSize(18).fontWeight(FontWeight.Medium)
                                Text(`编号：${item.code}`).fontSize(14).fontColor('#666')
                                Text(`价格：¥${item.price}`).fontSize(14).fontColor('#0078D4')
                            }
                            .padding(15)
                            .width('90%')
                            .backgroundColor('#F5F9FF')
                            .borderRadius(16)
                            .margin({ bottom: 10 })
                            .onClick(() => this.goToDetail(item))
                        }
                    }, (item: Asset) => item.id)
                }
                .width('100%')
                .layoutWeight(1)
            }
        }
        .width('100%')
        .height('100%')
    }
}

    🔰 阶段五：资产详情页（10分钟）

    📄 AssetDetailPage.ets 核心代码参考：

    import router from '@ohos.router';
import { Asset } from '../model/AssetModel';

@Entry
@Component
struct AssetDetailPage {
    @State asset: Asset | null = null;

    aboutToAppear() {
        const params = router.getParams() as Record;
        if (params && params['assetDetail']) {
            this.asset = params['assetDetail'] as Asset;
        }
    }

    build() {
        Column({ space: 20 }) {
            if (this.asset) {
                Text('🔍 资产详情').fontSize(24).fontWeight(FontWeight.Bold).margin({ top: 20 })
                Column({ space: 12 }) {
                    Text(`资产名称：${this.asset.name}`).fontSize(18)
                    Text(`资产编号：${this.asset.code}`).fontSize(16)
                    Text(`购入日期：${this.asset.date}`).fontSize(16)
                    Text(`价格：¥${this.asset.price}`).fontSize(18).fontColor('#E67E22')
                }
                .padding(20)
                .backgroundColor('#F8F8F8')
                .borderRadius(20)
                .width('90%')
                
                Button('返回列表', { type: ButtonType.Capsule })
                    .width('60%')
                    .onClick(() => {
                        router.back();
                    })
            } else {
                Text('未获取到资产信息').margin({ top: 50 })
            }
            Blank().layoutWeight(1)
        }
        .width('100%')
        .height('100%')
        .padding(20)
    }
}
