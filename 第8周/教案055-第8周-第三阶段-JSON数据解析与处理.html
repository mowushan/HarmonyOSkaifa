🧩 知识讲解 · JSON：互联网的通用语言

1. 什么是JSON？为什么重要？

📦 趣味引入： 你网购了一个包裹，快递单上写着“收件人：张三，物品：键盘”。JSON就像这张快递单，用简单的文字描述数据结构。

JSON语法规则（一眼就会）：

  • 数据在键值对中："name": "华为Mate60"
  • 键必须用双引号包裹，值可以是字符串、数字、布尔、数组、对象。
  • 对象用花括号 { }，数组用方括号 [ ]。
  • 多个键值对用逗号分隔。
{
  "code": 200,
  "message": "success",
  "data": {
    "total": 3,
    "assets": [
      {"id": 1, "name": "笔记本电脑", "price": 5999},
      {"id": 2, "name": "投影仪", "price": 3200}
    ]
  }
}

📢 几乎所有现代API都用JSON，学会它就能对接天气、新闻、资产管理系统等各种数据源。

2. 鸿蒙中如何解析JSON？

鸿蒙标准JavaScript/TypeScript语法，使用 JSON.parse(str) 把字符串转成对象，使用 JSON.stringify(obj) 把对象转成字符串。

// 模拟后端返回的JSON字符串
let jsonStr = '{"name":"鸿蒙手机","price":4999}';

// 解析为对象
let obj = JSON.parse(jsonStr);
console.info("商品名称：" + obj.name);   // 鸿蒙手机
console.info("价格：" + obj.price);      // 4999

// 反过来：对象转字符串
let product = { productId: 101, title: "智能手表" };
let str = JSON.stringify(product);
console.info(str);   // {"productId":101,"title":"智能手表"}

⚠️ 如果JSON格式错误（比如漏了引号），JSON.parse 会抛出异常，导致应用闪退。所以一定要用 try...catch 包裹！

🧪 实训任务一：手动解析JSON字符串（15分钟）

  • 定义JSON测试数据： 在页面中定义一个常量字符串，内容为资产数组的JSON。
  • 编写解析代码： 使用JSON.parse，遍历数组，把每条资产打印到控制台。
  • 异常保护： 故意修改JSON为错误格式，观察try-catch捕获错误。
    • @Entry
@Component
struct JsonDemo {
  @State assetList: string[] = [];

  aboutToAppear() {
    this.parseLocalJson();
  }

  parseLocalJson() {
    // 模拟从网络返回的JSON字符串
    let jsonString = `[
      {"id":1,"name":"ThinkPad笔记本","price":6800},
      {"id":2,"name":"鸿蒙开发板","price":1288},
      {"id":3,"name":"无线鼠标","price":99}
    ]`;

    try {
      let assets = JSON.parse(jsonString);
      let names: string[] = [];
      for (let i = 0; i < assets.length; i++) {
        names.push(assets[i].name + " - ¥" + assets[i].price);
      }
      this.assetList = names;
      console.info("解析成功，共" + assets.length + "项");
    } catch (error) {
      console.error("JSON解析失败：" + JSON.stringify(error));
    }
  }

  build() {
    Column() {
      Text("资产列表（本地JSON）").fontSize(20).margin(10);
      List() {
        ForEach(this.assetList, (item: string) => {
          ListItem() {
            Text(item).padding(10).fontSize(16);
          }
        })
      }
    }.width('100%').height('100%');
  }
}

    🧪 实训任务二：发起HTTP请求，解析真实API（25分钟）

  • 导入http模块： 在文件头部加上 import http from '@ohos.net.http';
  • 创建http请求： 在aboutToAppear中调用一个异步方法 fetchAssetsFromApi()。
  • 发送GET请求： 使用 http.createHttp().request()，指定URL。
  • 解析响应： 从响应中拿到result字段（JSON字符串），再解析并更新UI。
  • 处理loading状态： 请求中显示“加载中…”，加载完显示数据。
    • import http from '@ohos.net.http';

@Entry
@Component
struct JsonApiDemo {
  @State assets: AssetItem[] = [];
  @State isLoading: boolean = true;

  async fetchAssetsFromApi() {
    let httpRequest = http.createHttp();
    let url = "https://mockapi.hm-train.com/assets";  // 老师提供的模拟接口

    try {
      let response = await httpRequest.request(url, {
        method: http.RequestMethod.GET,
        connectTimeout: 60000,
        readTimeout: 60000
      });
      if (response.responseCode === 200) {
        let jsonStr = response.result as string;
        let result = JSON.parse(jsonStr);
        // 假设后端返回格式：{ "code":200, "data": [ {...} ] }
        if (result.code === 200) {
          this.assets = result.data;
        } else {
          console.error("接口返回错误：" + result.message);
        }
      } else {
        console.error(`请求失败，状态码：${response.responseCode}`);
      }
    } catch (err) {
      console.error("网络请求异常：" + JSON.stringify(err));
    } finally {
      this.isLoading = false;
      httpRequest.destroy();
    }
  }

  aboutToAppear() {
    this.fetchAssetsFromApi();
  }

  build() {
    Column() {
      if (this.isLoading) {
        LoadingProgress().width(50).height(50);
        Text("正在加载资产数据...").margin(10);
      } else {
        List() {
          ForEach(this.assets, (item: AssetItem) => {
            ListItem() {
              Row() {
                Text(item.name).fontSize(16).fontWeight(FontWeight.Medium)
                Blank()
                Text(`¥${item.price}`).fontColor('#f5a623')
              }.width('100%').padding(12)
            }
          })
        }
      }
    }.width('100%').height('100%').padding(16);
  }
}

interface AssetItem {
  id: number;
  name: string;
  price: number;
}
